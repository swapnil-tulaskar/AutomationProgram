Q) How many ways we can reusing existing class members in new class?
A) in 2 ways	
	1. by using HAS-A relation (Association)
	2. by using IS-A relation (Inheritance)

	- HAS-A relation means 
		creating existing class object in new class
		and accessing members by using the existing class object
	
	- IS-A relation	means
		creating new class deriving from an existing class
		and accessing existing class members by using this new class object

- Below code explains access Person class members in Student class for Person object
  with HAS-A relation( Person class members are not executing for Student object)

	class Person {
		Sting name;

		void eat() {
			System.out.println(name + " is eating");
		}
	}
//=====================================================================

	class Student {  
		public static void main(String[] args) {
		  //=== HAS-A === realation wise reusing
			Person p1 = new Person();//Student class is reusing
			p1.name = "Hari";	//Person class and its memebers
			p1.eat();		//for perform operation for Person
		}
	}
	//with HAS-A relation we are reusing 
	//Person and its members for Person but not student

	//for using Person members for Student, 
	//we must develop IS-A realtion(Inheritance)

	//By developing inheritance we are 
	//'extending and reusing' existing class members
	//for new class object as if those existing members
	//available in this new class
//=====================================================================

	class Student extends Person { //extending Person to use as Student

		public static void main(String[] args) {
		  //reusing with IS-A relation	
			Student s1 = new Student();	
			s1.name = "hari";	//we are accessing and reusing
			s1.eat();		//existing class Person members
						//as if those are own members of 
		}				//Student object 
	}


	//Inheritance is meant for 'extending + resuing'

//Person.java
class Person {
	String name;

	void eat(){
		System.out.println(name + " is eating");
	}
	void sleep() {
		System.out.println(name + " is deep sleeping");
	}
}

//Student.java
class Student extends Person {
	int sno;	 //extending		
	String course;

	void listen() {		  //reusing Person class members in Student
	    System.out.println(name + " is listening "+ course);
	}			//as if it is Student class member

	void reply() {
	    System.out.println(name + " is replying to "+ course + " questions");
	}

	void write() {
	    System.out.println(name + " is writing "+ course + " notes");
	}

	void read() {
	    System.out.println(name + " is reading "+ course + " notes");
	}
}
